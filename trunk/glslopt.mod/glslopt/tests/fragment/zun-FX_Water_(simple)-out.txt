    (normal_2.x * normal_2.x)
  ) - (normal_2.y * normal_2.y)));
  vec4 normal_3;
  normal_3.xy = ((texture2D (_BumpMap, gl_TexCoord[1].xy).wy * 2.0) - 1.0);
  normal_3.z = sqrt(((1.0 -

  ) - (normal_3.y * normal_3.y)));

}
